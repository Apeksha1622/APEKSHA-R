C++ Code Example

The following is a C++ code example:


#include <iostream>
using namespace std;

// Function to display the elements of the array
void displayArray(int arr[], int size) {
    if (size == 0) {
        cout << "Array is empty." << endl;
    } else {
        cout << "Current Array: ";
        for (int i = 0; i < size; i++) {
            cout << arr[i] << " ";
        }
        cout << endl;
    }
}

int main() {
    const int MAX_SIZE = 100; // Maximum size of the array
    int arr[MAX_SIZE];        // Declare an array
    int size = 0;             // Current number of elements in the array

    int choice;
    do {
        // Display the menu
        cout << "\nMenu:\n";
        cout << "1. Add an element\n";
        cout << "2. Remove the last element\n";
        cout << "3. Display the array\n";
        cout << "4. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1: {
                if (size < MAX_SIZE) {
                    int value;
                    cout << "Enter the value to add: ";
                    cin >> value;
                    arr[size] = value; // Add the value to the array
                    size++;            // Increment the size
                } else {
                    cout << "Array is full. Cannot add more elements." << endl;
                }
                break;
            }
            case 2: {
                if (size > 0) {
                    size--; // Decrease the size to "remove" the last element
                    cout << "Last element removed." << endl;
                } else {
                    cout << "Array is empty. Nothing to remove." << endl;
                }
                break;
            }
            case 3: {
                displayArray(arr, size); // Display the current array
                break;
            }
            case 4: {
                cout << "Exiting program." << endl;
                break;
            }
            default: {
                cout << "Invalid choice. Please try again." << endl;
            }
        }
    } while (choice != 4);

    return 0;
}
